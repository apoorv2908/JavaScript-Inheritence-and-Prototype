Question 1: Create 2 objects - parent and child. Attach methods to parent and use those methods in child object using parents prototype

        Code

        const parent ={
            details(){
                console.log(`My name is ${this.name} and I am from ${this.location}`)
            }
            }
            const firstchild = Object.create(parent);
            firstchild.name ="apoorv";
            firstchild.location="delhi";
            
            firstchild.details();
            
            const secondChild= Object.create(parent)
            secondChild.name="arjun"
            secondChild.location="delhi"
            
            secondChild.details()

        My name is apoorv and I am from delhi
        My name is arjun and I am from delhi
